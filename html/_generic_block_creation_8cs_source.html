50 public static void CreateFunctionBlock(string blockName, string instanceDbName,
51 ProgrammingLanguage programmingLanguage,
52 PlcDevice plcDevice)
53 {
54 // Input validation
55 if (string.IsNullOrEmpty(blockName)) throw new ArgumentNullException(nameof(blockName));
56 if (string.IsNullOrEmpty(instanceDbName)) throw new ArgumentNullException(nameof(instanceDbName));
57 if (plcDevice == null) throw new ArgumentNullException(nameof(plcDevice));
58 if (programmingLanguage != ProgrammingLanguage.LAD && programmingLanguage != ProgrammingLanguage.FBD &&
59 programmingLanguage != ProgrammingLanguage.SCL)
60 throw new ArgumentException(nameof(programmingLanguage));
61
62 // Create block call with default values
63 var fbCall = new BlockCall(instanceDbName, plcDevice) { ["Input1"] = "1", ["Input2"] = "0" };
64
65 // Create and configure network
66 var fbNetwork = new BlockNetwork { NetworkTitles = { [TypeMapper.BaseCulture.Name] = "myFB Network Title" } };
67 fbNetwork.Blocks.Add(fbCall);
68
69 // Create and Configure FB properties
70 var fb = new XmlFB(blockName);
71 fb.Networks.Add(fbNetwork);
72 fb.BlockAttributes.ProgrammingLanguage = programmingLanguage;
73 fb.BlockComments[TypeMapper.BaseCulture.Name] = "myFB Block Comment";
74 fb.BlockTitles[TypeMapper.BaseCulture.Name] = "myFB Block Title";
75
76 // Generate the block in appropriate format
77 if (programmingLanguage == ProgrammingLanguage.SCL)
78 fb.GenerateSclBlock(plcDevice);
79 else
80 fb.GenerateXmlBlock(plcDevice);
81 }

266 {
267 var mainOb = new XmlOB("Main");
268 mainOb.BlockAttributes.ProgrammingLanguage = ProgrammingLanguage.LAD;

270
271 var itf = mainOb.Interface[InterfaceSections.Temp];
272 var temp = new InterfaceParameter("Temp", "Bool");
273 var tempInt = new InterfaceParameter("TempInt", "Int");
274 itf.Add(temp);
275 itf.Add(tempInt);
276
277 //create block call of instance DB
278 var myFBCall = new BlockCall(instanceDbName, plcDevice)
279 {
280 ["Input1"] = "#Temp", // inputVariable1
281 ["Input2"] = "0" // inputVariable1
282 //["Input2"] = "inputVariable1",
283
284 //["Output1"] = "outputVariable1",
285 };
286 var blockNetwork = new BlockNetwork();
287 blockNetwork.Blocks.Add(myFBCall);
288 blockNetwork.GenerationLabel =
289 new GenerationLabel(module.Name, "Generated by MAC", module.ModuleID.ToString());
290 mainOb.Networks.Add(blockNetwork);
291
293 var content = "\"myDB\".myParameterName := 100;\n";
294 content += "IF #Temp THEN\n";
295 content += "#TempInt := 5;\n";
296 content += "ELSE\n";
297 content += "TempInt := 1;\n";
298 content += "END_IF;\n\n";
299 var sclContent = content;
300 var sclcode_network = ParseSingleSCLCall(sclContent, plcDevice, mainOb);
301 sclcode_network.GenerationLabel =
302 new GenerationLabel(module.Name, "Virtual_Master_scl_code", module.ModuleID.ToString());
303 mainOb.Networks.Add(sclcode_network);

316 public static void AddCodeBlockToOB(XmlLogicalBlock codeBlock, ResourceManagement resourceManagement,
317 string languageSettings, PlcDevice plcDevice)
318 {
319 var existingOB = OpennessFuncs.TryGetOBFromPlcDevice(plcDevice, codeBlock.Name);
320 //Create new OB in SCL with calling code
321 if (existingOB == null)
322 {
323 if (codeBlock.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
324 codeBlock.GenerateSclBlock(plcDevice);
325 else //insert networks to KOP/FUP
326 codeBlock.GenerateXmlBlock(plcDevice, null, languageSettings);
327 }
328 else
329 //Insert Code to existing OB
330 {
331 var blockAsDoc =
332 OpennessFuncs.ExportBlockAsXml(existingOB.Name, plcDevice, OpennessFuncs.CachingOptions.DISABLED);
333 var parser = new Parser();
334 var xmlOB = parser.ParseXml(blockAsDoc);
335 xmlOB.MergeBlock(codeBlock);
336 //add SCL-Code
337 if (xmlOB.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
338 xmlOB.GenerateSclBlock(plcDevice);
339 else //insert networks to KOP/FUP
340 xmlOB.GenerateXmlBlock(plcDevice, null, languageSettings);
341 }

381 private static INetwork ParseSingleSCLCall(string sclCall, PlcDevice plc, XmlBlock xmlBlock)
382 {
383 var parsed = new Parser().ParseSclSnippet(sclCall, xmlBlock, plc, GroupBlockCalls.NOGROUPING)
384 .FirstOrDefault();
385 if (parsed is BlockNetwork blockNetwork) return blockNetwork;
386
387 if (parsed is FixNetwork fixNetwork) return fixNetwork;
388 return null;
389 }
