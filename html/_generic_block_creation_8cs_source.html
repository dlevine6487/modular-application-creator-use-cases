51 public static void CreateFunctionBlock(string blockName, string instanceDbName,
52 ProgrammingLanguage programmingLanguage,
53 PlcDevice plcDevice)
54 {
55 // Input validation
56 if (string.IsNullOrEmpty(blockName)) throw new ArgumentNullException(nameof(blockName));
57 if (string.IsNullOrEmpty(instanceDbName)) throw new ArgumentNullException(nameof(instanceDbName));
58 if (plcDevice == null) throw new ArgumentNullException(nameof(plcDevice));
59 if (programmingLanguage != ProgrammingLanguage.LAD && programmingLanguage != ProgrammingLanguage.FBD &&
60 programmingLanguage != ProgrammingLanguage.SCL)
61 throw new ArgumentException(nameof(programmingLanguage));
62
63 // Create block call with default values
64 var fbCall = new BlockCall(instanceDbName, plcDevice) { ["Input1"] = "1", ["Input2"] = "0" };
65
66 // Create and configure network
67 var fbNetwork = new BlockNetwork { NetworkTitles = { [TypeMapper.BaseCulture.Name] = "myFB Network Title" } };
68 fbNetwork.Blocks.Add(fbCall);
69
70 // Create and Configure FB properties
71 var fb = new XmlFB(blockName);
72 fb.Networks.Add(fbNetwork);
73 fb.BlockAttributes.ProgrammingLanguage = programmingLanguage;
74 fb.BlockComments[TypeMapper.BaseCulture.Name] = "myFB Block Comment";
75 fb.BlockTitles[TypeMapper.BaseCulture.Name] = "myFB Block Title";
76
77 // Generate the block in appropriate format
78 if (programmingLanguage == ProgrammingLanguage.SCL)
79 fb.GenerateSclBlock(plcDevice);
80 else
81 fb.GenerateXmlBlock(plcDevice);
82 }

309 public static void GenerateOB_Main(string instanceDbName, MAC_use_casesEM module, string languageSettings,
310 PlcDevice plcDevice)
311 {
312 var mainOb = new XmlOB("Main");
313 mainOb.BlockAttributes.ProgrammingLanguage = ProgrammingLanguage.LAD;
314 ((OB_BlockAttributes)mainOb.BlockAttributes).BlockSecondaryType = "ProgramCycle";
315
316 var itf = mainOb.Interface[InterfaceSections.Temp];
317 var temp = new InterfaceParameter("Temp", "Bool");
318 var tempInt = new InterfaceParameter("TempInt", "Int");
319 itf.Add(temp);
320 itf.Add(tempInt);
321
322 //create block call of instance DB
323 var myFBCall = new BlockCall(instanceDbName, plcDevice)
324 {
325 ["Input1"] = "#Temp", // inputVariable1
326 ["Input2"] = "0" // inputVariable1
327 //["Input2"] = "inputVariable1",
328
329 //["Output1"] = "outputVariable1",
330 };
331 var blockNetwork = new BlockNetwork();
332 blockNetwork.Blocks.Add(myFBCall);
333 blockNetwork.GenerationLabel =
334 new GenerationLabel(module.Name, "Generated by MAC", module.ModuleID.ToString());
335 mainOb.Networks.Add(blockNetwork);
336
338 var content = "\"myDB\".myParameterName := 100;\n";
339 content += "IF #Temp THEN\n";
340 content += "#TempInt := 5;\n";
341 content += "ELSE\n";
342 content += "TempInt := 1;\n";
343 content += "END_IF;\n\n";
344 var sclContent = content;
345 var sclcode_network = ParseSingleSCLCall(sclContent, plcDevice, mainOb);
346 sclcode_network.GenerationLabel =
347 new GenerationLabel(module.Name, "Virtual_Master_scl_code", module.ModuleID.ToString());
348 mainOb.Networks.Add(sclcode_network);

361 public static void AddCodeBlockToOB(XmlLogicalBlock codeBlock, ResourceManagement resourceManagement,
362 string languageSettings, PlcDevice plcDevice)
363 {
364 var existingOB = OpennessFuncs.TryGetOBFromPlcDevice(plcDevice, codeBlock.Name);
365 //Create new OB in SCL with calling code
366 if (existingOB == null)
367 {
368 if (codeBlock.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
369 codeBlock.GenerateSclBlock(plcDevice);
370 else //insert networks to KOP/FUP
371 codeBlock.GenerateXmlBlock(plcDevice, null, languageSettings);
372 }
373 else
374 //Insert Code to existing OB
375 {
376 var blockAsDoc =
377 OpennessFuncs.ExportBlockAsXml(existingOB.Name, plcDevice, OpennessFuncs.CachingOptions.DISABLED);
378 var parser = new Parser();
379 var xmlOB = parser.ParseXml(blockAsDoc);
380 xmlOB.MergeBlock(codeBlock);
381 //add SCL-Code
382 if (xmlOB.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
383 xmlOB.GenerateSclBlock(plcDevice);
384 else //insert networks to KOP/FUP
385 xmlOB.GenerateXmlBlock(plcDevice, null, languageSettings);
386 }

426 private static INetwork ParseSingleSCLCall(string sclCall, PlcDevice plc, XmlBlock xmlBlock)
427 {
428 var parsed = new Parser().ParseSclSnippet(sclCall, xmlBlock, plc, GroupBlockCalls.NOGROUPING)
429 .FirstOrDefault();
430 if (parsed is BlockNetwork blockNetwork) return blockNetwork;
431
432 if (parsed is FixNetwork fixNetwork) return fixNetwork;
433 return null;
434 }
