198 var mainOb = new XmlOB("Main");

201
202 var itf = mainOb.Interface[InterfaceSections.Temp];

204 var tempInt = new InterfaceParameter("TempInt", "Int");
205 itf.Add(temp);
206 itf.Add(tempInt);
207
208 //create block call of instance DB
209 var myFBCall = new BlockCall(instanceDbName, plcDevice)
210 {
211 ["Input1"] = "#Temp", // inputVariable1
212 ["Input2"] = "0" // inputVariable1

214
215 //["Output1"] = "outputVariable1",
216 };
217 var blockNetwork = new BlockNetwork();
218 blockNetwork.Blocks.Add(myFBCall);
219 blockNetwork.GenerationLabel =

221 mainOb.Networks.Add(blockNetwork);
222
224 var content = "\"myDB\".myParameterName := 100;\n";
225 content += "IF #Temp THEN\n";
226 content += "#TempInt := 5;\n";
227 content += "ELSE\n";
228 content += "TempInt := 1;\n";
229 content += "END_IF;\n\n";
230 var sclContent = content;
231 var sclcode_network = ParseSingleSCLCall(sclContent, plcDevice, mainOb);

247 public static void AddCodeBlockToOB(XmlLogicalBlock codeBlock, ResourceManagement resourceManagement,
248 string languageSettings, PlcDevice plcDevice)
249 {
250 var existingOB = OpennessFuncs.TryGetOBFromPlcDevice(plcDevice, codeBlock.Name);
251 //Create new OB in SCL with calling code
252 if (existingOB == null)
253 {
254 if (codeBlock.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
255 codeBlock.GenerateSclBlock(plcDevice);
256 else //insert networks to KOP/FUP
257 codeBlock.GenerateXmlBlock(plcDevice, null, languageSettings);
258 }
259 else
260 //Insert Code to existing OB
261 {
262 var blockAsDoc =
263 OpennessFuncs.ExportBlockAsXml(existingOB.Name, plcDevice, OpennessFuncs.CachingOptions.DISABLED);
264 var parser = new Parser();
265 var xmlOB = parser.ParseXml(blockAsDoc);
266 xmlOB.MergeBlock(codeBlock);
267 //add SCL-Code
268 if (xmlOB.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
269 xmlOB.GenerateSclBlock(plcDevice);
270 else //insert networks to KOP/FUP
271 xmlOB.GenerateXmlBlock(plcDevice, null, languageSettings);
272 }

312 private static INetwork ParseSingleSCLCall(string sclCall, PlcDevice plc, XmlBlock xmlBlock)
313 {
314 var parsed = new Parser().ParseSclSnippet(sclCall, xmlBlock, plc, GroupBlockCalls.NOGROUPING)
315 .FirstOrDefault();
316 if (parsed is BlockNetwork blockNetwork) return blockNetwork;
317
318 if (parsed is FixNetwork fixNetwork) return fixNetwork;
319 return null;
320 }
